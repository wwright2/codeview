int x;
cin>>x;
int answer = 1;  //initiliaze by one (not 0 because then the result would always be 0)
for (int i=2; i<=x; i++){
    answer = answer*i;    //multiply answer by all the numbers from 2 two x
}
cout<<answer;
}
