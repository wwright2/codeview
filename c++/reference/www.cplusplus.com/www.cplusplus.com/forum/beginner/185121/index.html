#include <iostream>

using namespace std;

int main(int argc, char **argv)
{
  cout << "Press 1 to see Hello world or press 2 to see Hello user. ";

  char ch;
  if(cin.get(ch))
  {
    if (ch == '1')
      cout << "Hello world.\n\n";
    else if (ch == '2')
      cout << "Hello User\n\n";
  }    
  system("PAUSE");

  return 0;
}
