// Generate the next world
void generate(int currentboard[][BOARDCOLS], int nextboard[][BOARDCOLS])
{
    // http://pmav.eu/stuff/javascript-game-of-life-v3.1.1/
    /**
    Any live cell with fewer than two live neighbors dies, as if caused by under-population.
    Any live cell with two or three live neighbors lives on to the next generation.
    Any live cell with more than three live neighbors dies, as if by over-population.
    Any dead cell with exactly three live neighbors becomes a live cell, as if by reproduction.
    */
    // iterate over currentboard
    //   call checkneighbors with current cell and current row and column
    //   apply rules and use the result to add either a new cell or leave the current cell alone in the next board

    
    int neighbors = checkneighbors(currentboard);
    for (int r = 0; r < BOARDROWS; r++)
    {
        for (int c = 0; c < BOARDCOLS; c++)
        {
            if (currentboard[r][c] == LIVE && neighbors < 2)
            {
                nextboard[r][c] = DEAD;
            }
            else if (currentboard[r][c] == LIVE && (neighbors == 2 || neighbors == 3))
            {
                nextboard[r][c] = LIVE;
            }
            else if (currentboard[r][c] == LIVE && neighbors > 3)
            {
                nextboard[r][c] = DEAD;
            }
            else if (currentboard[r][c] == DEAD && neighbors == 3)
            {
                nextboard[r][c] = LIVE;
            }
        }
    }

}

// check the neighbors
//will check each 8 neighbors and add to the counter if it spots a 1
//will return the final counter
int checkneighbors(int board[][BOARDCOLS])
{
    int neighbor_count = 0;
    for (int r = 0; r < BOARDROWS; r++)
    {
        for (int c=0; c < BOARDCOLS; c++)
        {
            if (board[r][c-1] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r-1][c-1] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r-1][c] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r-1][c+1] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r][c+1] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r+1][c+1] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r+1][c] == LIVE)
            {
                neighbor_count++;
            }
            else if (board[r+1][c-1] == LIVE)
            {
                neighbor_count++;
            }
        }
    }
    return neighbor_count;
}

// copy one board to another
void copy(int source[][BOARDCOLS], int dest[][BOARDCOLS])
{
    for (int r = 0; r < BOARDROWS; r++)
    {
        for (int c = 0; c < BOARDCOLS; c++)
        {
            dest[r][c] = source[r][c];
        }
    }
}

// display the board
void display(int board[][BOARDCOLS])
{
    for (int r = 0; r < BOARDROWS; r++)
    {
        for (int c = 0; c < BOARDCOLS; c++)
        {
            cout << board[r][c];
        }
    } cout << endl;
}
