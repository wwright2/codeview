#include <iostream>
#include <fstream>
#include <vector>
#include <string>

using namespace std;

void readLine(vector<string> &playlist);

void main(void)
{
  vector<string> input;
  
  readLine(input);
  for (int i=0; i < input.size(); i++)
    cout << input[i] << endl; 

  system("pause");
}

void readLine(vector<string> &playlist)
{
   ifstream src("Playlist.txt");

  if (!src)
  {
    cerr << "Error opening file";
    exit(EXIT_FAILURE);
  }

  string buffer;
 
  while (src)
  {
    getline(src, buffer);
    playlist.push_back(buffer);
  }
}
