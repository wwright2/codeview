#include <iostream>
#include <string>
#include <vector>
using namespace std;

vector<string> names;
int Play = 1;

void check_name (const string & input)
{   for(size_t i = 0; i < names.size(); i++)
    {   if (input == names[i])
        {   Play = 0;
        }
    }
}

int main()
{   string Input;
    
    cout << "Welcome to the Fake Name Generator Game" << endl;
    cout << "Press <Any> Key To Start" << endl;
    cin >> Input;
    do
    {   cout << "Type A New Name: " << flush;
        cin >> Input;
        names.push_back (Input); 
        check_name(Input);
    } while (Play == 1);
    cout << "Game Over" << endl;
    system ("pause"); 
}
