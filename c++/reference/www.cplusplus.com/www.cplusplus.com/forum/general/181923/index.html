#include <cstdint>
#include <iomanip>
#include <iostream>
#include <sstream>
#include <string>

int main()
{
    int i = 5;

    std::cout << &i << '\n';

    void* ptr = &i;
    intptr_t as_int = reinterpret_cast<std::intptr_t>(ptr);

    std::ostringstream oss;
    oss << std::hex << std::showbase << as_int;
    std::cout << oss.str() << '\n';

    as_int = 0;
    ptr = nullptr;

    as_int = std::stoll(oss.str(), nullptr, 16);

    ptr = reinterpret_cast<void*>(as_int);

    std::cout << ptr << '\n';
}
