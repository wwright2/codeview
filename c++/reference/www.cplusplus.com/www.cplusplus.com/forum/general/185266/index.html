class ti
{
public:
	vector<double> in;
	ti(int n, ...)
	{
		va_list vl;
		va_start(vl, n);
		for (int i = 0; i<n; i++)
		{
			in.push_back(va_arg(vl, double));
		}
		va_end(vl);
	}
};

class tr
{
public:
	vector<double> results;
	tr(int n, ...)
	{
		va_list vl;
		va_start(vl, n);
		for (int i = 0; i<n; i++)
		{
			results.push_back(va_arg(vl, double));
		}
		va_end(vl);
	}
};
